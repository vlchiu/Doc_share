Ngoài ra có 1 lưu ý nữa là sau khi đồng bộ copy source từ bên vendor/hsptek sang source chính thì ae cần thay đổi các file có đuôi .mk, .bp, .bb, thành đuôi _mk, _bp, _bb, để tránh lúc build hệ thống nó load cả 2 nơi dẫn đến lỗi conflict.

cd vendor/hsptek 

for x in $(find . -name \*.mk); do
mv $x $(echo "$x" | sed 's/\.mk$/_mk/')
done

for x in $(find . -name \*.bp); do
mv $x $(echo "$x" | sed 's/\.bp$/_bp/')
done

for x in $(find . -name \*.bb); do
mv $x $(echo "$x" | sed 's/\.bb$/_bb/')
done
